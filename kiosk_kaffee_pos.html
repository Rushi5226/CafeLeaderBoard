Copy Web App URL → paste below function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName("Orders") || ss.insertSheet("Orders"); if(sh.getLastRow()===0) sh.appendRow(["Order#","Date","Time","Items","Note","Total","Status"]); var d = JSON.parse(e.postData.contents); sh.appendRow([d.orderNo,d.date,d.time,d.items,d.note,d.total,d.status]); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); }
